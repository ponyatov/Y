(\file{mingw32.?pp}\ files).

\secrel{Display}
\secrel{Window}

Window object controls windows in GUI environments

\begin{verbatim}
window @ Sym -> window:sym.val	# window constructor using name from Sym.val
class:Window
	Window @ Sym -> window:sym.val		# constructor: crerate (hided) window
	fn:hide								# hide window
	fn:show								# show window (hided by default)
	fn:minimize
	fn:fullscreen
\end{verbatim}
